// ── Data helpers ──────────────────────────────────────────────────────────────

const HOLES = 18, PAR = 2;

const SS = (s) => s === 0 ? 'E' : s > 0 ? '+'+s : ''+s;

const totalScore = (h) => h.reduce((s,v) => s + (v != null ? v - PAR : 0), 0);

const scoreInfo = (p) => {

  if(p===1) return {label:'Birdie',color:'#f59e0b',symbol:'🐦'};

  if(p===2) return {label:'Par',color:'#22c55e',symbol:'✓'};

  if(p===3) return {label:'Bogey',color:'#ef4444',symbol:'+1'};

  if(p>=4)  return {label:`+${p-2}`,color:'#dc2626',symbol:`+${p-2}`};

  return {label:'-',color:'#6b7280',symbol:'-'};

};

const loadRounds = () => { try{return JSON.parse(localStorage.getItem('pt_rounds')||'[]')}catch{return[]} };

const saveRounds = (r) => { try{localStorage.setItem('pt_rounds',JSON.stringify(r))}catch{} };

const fmtDate = (d) => new Date(d).toLocaleDateString('en-GB',{weekday:'short',day:'numeric',month:'short',year:'numeric'});

const fmtShort = (d) => new Date(d).toLocaleDateString('en-GB',{day:'numeric',month:'short'});


// ── State ─────────────────────────────────────────────────────────────────────

let state = {

  view: 'home',

  rounds: loadRounds(),

  round: null, // {twoPlayer, names, holes[[]], activePlayer, currentHole, finished}

};


function setState(updates) {

  Object.assign(state, updates);

  render();

}


// ── Canvas chart ──────────────────────────────────────────────────────────────

function drawChart(canvasId, data) {

  requestAnimationFrame(() => {

    const canvas = document.getElementById(canvasId);

    if(!canvas) return;

    const dpr = window.devicePixelRatio||1;

    const w = canvas.offsetWidth, h = canvas.offsetHeight;

    canvas.width = w*dpr; canvas.height = h*dpr;

    const ctx = canvas.getContext('2d');

    ctx.scale(dpr,dpr);

    const pad = {t:10,r:10,b:30,l:36};

    const cw = w-pad.l-pad.r, ch = h-pad.t-pad.b;

    const scores = data.map(d=>d.score);

    const minS = Math.min(...scores,-1)-1, maxS = Math.max(...scores,1)+1;

    const range = maxS-minS||1;

    const xStep = data.length > 1 ? cw/(data.length-1) : cw;

    const toY = (s) => pad.t + ch - ((s-minS)/range)*ch;

    const toX = (i) => pad.l + (data.length>1 ? i*xStep : cw/2);

    // grid

    ctx.strokeStyle='#374151'; ctx.lineWidth=1;

    for(let s=Math.ceil(minS);s<=Math.floor(maxS);s++){

      const y=toY(s);

      ctx.beginPath(); ctx.moveTo(pad.l,y); ctx.lineTo(pad.l+cw,y); ctx.stroke();

      ctx.fillStyle='#6b7280'; ctx.font='10px -apple-system,sans-serif';

      ctx.textAlign='right'; ctx.fillText(SS(s),pad.l-4,y+3);

    }

    // x labels

    ctx.fillStyle='#6b7280'; ctx.font='10px -apple-system,sans-serif'; ctx.textAlign='center';

    data.forEach((d,i)=>{ ctx.fillText(d.round,toX(i),h-8); });

    // line

    ctx.beginPath(); ctx.strokeStyle='#f59e0b'; ctx.lineWidth=2.5; ctx.lineJoin='round';

    data.forEach((d,i)=>{ i===0?ctx.moveTo(toX(i),toY(d.score)):ctx.lineTo(toX(i),toY(d.score)); });

    ctx.stroke();

    // dots

    data.forEach((d,i)=>{

      ctx.beginPath(); ctx.arc(toX(i),toY(d.score),5,0,Math.PI*2);

      ctx.fillStyle='#f59e0b'; ctx.fill();

    });

  });

}


// ── Render ────────────────────────────────────────────────────────────────────

function render() {

  const app = document.getElementById('app');

  const {view,rounds,round} = state;

  if(view==='home')    { app.innerHTML = renderHome(); }

  if(view==='setup')   { app.innerHTML = renderSetup(); }

  if(view==='round')   { app.innerHTML = renderRound(); }

  if(view==='history') { app.innerHTML = renderHistory(); }

  if(view==='stats')   { app.innerHTML = renderStats(); attachStatsListeners(); }

  attachListeners();

  // draw chart if on stats

  if(view==='stats') {

    const names = [...new Set(rounds.map(r=>r.player))];

    const sel = document.getElementById('selPlayer')?.value || names[0];

    if(sel) drawChartForPlayer(sel);

  }

}

// ── Round ─────────────────────────────────────────────────────────────────────

function renderRound() {

  const {round,rounds} = state;

  if(!round) return '';

  if(round.finished) return renderFinished();

  const {twoPlayer,names,holes,activePlayer,currentHole} = round;

  const curHoles = holes[activePlayer];

  const curPutts = curHoles[currentHole];

  const score = totalScore(curHoles);

  const isLast = currentHole===HOLES-1;

  const isLastAction = isLast && (!twoPlayer || activePlayer===1);

  return `

  <div class="screen">

    <div class="round-header">

      <button class="quit" onclick="if(confirm('Quit this round?'))setState({view:'home',round:null})">✕ Quit</button>

      <div class="hole-info">

        <div class="direction">${currentHole<9?'Clockwise':'Anti-Clockwise'}</div>

        <div class="hole-num">Hole ${currentHole+1}</div>

      </div>

      <div class="score-box">

        <div class="slabel">Score</div>

        <div class="sval" style="color:${score<0?'#f59e0b':score>0?'#ef4444':'#22c55e'}">${SS(score)}</div>

      </div>

    </div>

    ${twoPlayer?`

    <div class="player-tabs">

      ${names.map((n,i)=>`<div class="player-tab" style="background:${activePlayer===i?'#f59e0b':'#1f2937'};color:${activePlayer===i?'#111827':'#9ca3af'}">${n} ${SS(totalScore(holes[i]))}</div>`).join('')}

    </div>`:``}

    <div class="putt-area">

      <div class="putt-label">${twoPlayer?names[activePlayer]+' — ':''}How many putts?</div>

      <div class="putt-buttons">

        ${[1,2,3,4].map(n=>{

          const info=scoreInfo(n);

          const sel=curPutts===n;

          return `<button class="putt-btn" onclick="logPutts(${n})"

            style="background:${sel?info.color+'44':'#1f2937'};border-color:${sel?info.color:'#374151'};color:${sel?info.color:'#9ca3af'};transform:${sel?'scale(1.12)':'scale(1)'}">${n}</button>`;

        }).join('')}

      </div>

      ${curPutts?`<div class="putt-result"><div class="symbol">${scoreInfo(curPutts).symbol}</div><div class="result-label" style="color:${scoreInfo(curPutts).color}">${scoreInfo(curPutts).label}</div></div>`:''}

    </div>

    <div class="hole-grid">

      <div class="hole-grid-inner">

        ${Array(HOLES).fill(null).map((_,i)=>{

          const h=curHoles[i];

          const info=h?scoreInfo(h):null;

          const isCur=i===currentHole;

          return `<button class="hole-cell" onclick="jumpHole(${i})"

            style="background:${isCur?'#f59e0b22':info?info.color+'22':'#1f2937'};border-color:${isCur?'#f59e0b':info?info.color:'#374151'};color:${isCur?'#f59e0b':info?info.color:'#6b7280'}">${h||i+1}</button>`;

        }).join('')}

      </div>

      <div class="grid-labels"><span>← Clockwise (1-9)</span><span>Anti-CW (10-18) →</span></div>

    <div class="nav-row">

      <button class="btn-back" onclick="prevHole()" ${currentHole===0&&activePlayer===0?'disabled':''}>← Back</button>

      <button class="btn-next" onclick="nextHole()" ${curPutts==null?'disabled':''}>${isLastAction?'Finish 🏁':'Next →'}</button>

    </div>

  </div>`;

}


function logPutts(n) {

  const r = state.round;

  r.holes = r.holes.map((ph,i)=>i===r.activePlayer?ph.map((h,j)=>j===r.currentHole?n:h):ph);

  render();

}

function jumpHole(i) { state.round.currentHole=i; state.round.activePlayer=0; render(); }

function nextHole() {

  const r = state.round;

  if(r.holes[r.activePlayer][r.currentHole]==null) return;

  if(r.twoPlayer && r.activePlayer===0) { r.activePlayer=1; render(); return; }

  r.activePlayer=0;

  if(r.currentHole===HOLES-1) {

    r.finished=true;

    const saved = r.names.map((name,i)=>({id:Date.now()+i,player:name,holes:r.holes[i],score:totalScore(r.holes[i]),date:new Date().toISOString()}));

    const newRounds = [...state.rounds, ...saved];

    saveRounds(newRounds);

    state.rounds = newRounds;

    render();

  } else {

    r.currentHole++;

    render();

  }

}

function prevHole() {

  const r = state.round;

  if(r.twoPlayer && r.activePlayer===1) { r.activePlayer=0; render(); return; }

  if(r.twoPlayer) r.activePlayer=1;

  if(r.currentHole>0) r.currentHole--;

  render();

}

  const {names,holes,twoPlayer} = round;

      <div class="trophy">🏆</div>

      <h2>Round Complete!</h2>

      ${names.map((name,i)=>{

        const sc=totalScore(holes[i]);

        const pr=rounds.filter(r=>r.player===name);

        const isFirst=pr.length===1;

        return `<div class="result-card">

          <div class="pname">${name}</div>

          <div class="big-score" style="color:${sc<0?'#f59e0b':sc===0?'#22c55e':'#ef4444'}">${SS(sc)}</div>

          <div class="sub">${sc>=0?'+':''}${sc} vs par over 18 holes</div>

          ${isNewPB?'<div class="note" style="color:#f59e0b">🎉 New Personal Best!</div>':''}

          ${isFirst?'<div class="note" style="color:#22c55e">First round recorded!</div>':''}

          ${sc<=-6?'<div class="note" style="color:#c084fc">🔥 You matched the record!</div>':''}

        </div>`;

      }).join('')}

        <p>Head-to-Head</p>

        <div class="winner">${totalScore(holes[0])<totalScore(holes[1])?'🏆 '+names[0]+' wins!':totalScore(holes[1])<totalScore(holes[0])?'🏆 '+names[1]+' wins!':"It's a tie! 🤝"}</div>

      </div>`:``}

      <button class="btn-done" onclick="setState({view:'home',round:null})">Done</button>

    </div>

  </div>`;

}

      <h2>Round History</h2>

function getStats(name) {

  const rs = state.rounds.filter(r=>r.player===name);

  if(!rs.length) return null;

  const allHoles = rs.flatMap(r=>r.holes.filter(h=>h!=null));

  const birdies=allHoles.filter(h=>h===1).length;

  const pars=allHoles.filter(h=>h===2).length;

  const bogeys=allHoles.filter(h=>h===3).length;

  const doubles=allHoles.filter(h=>h>=4).length;

  return {

    rounds:rs.length,

    avg:(rs.reduce((s,r)=>s+r.score,0)/rs.length).toFixed(1),

    best:Math.min(...rs.map(r=>r.score)),

    worst:Math.max(...rs.map(r=>r.score)),

    avgBirdies:(birdies/rs.length).toFixed(1),

    avgPars:(pars/rs.length).toFixed(1),

    avgBogeys:(bogeys/rs.length).toFixed(1),

    avgDoubles:(doubles/rs.length).toFixed(1),

    birdieRate:allHoles.length?((birdies/allHoles.length)*100).toFixed(0):0,

    chartData:rs.map((r,i)=>({round:i+1,score:r.score})),

  };

}

function drawChartForPlayer(name) {

  const stats = getStats(name);

  if(stats && stats.chartData.length>1) drawChart('scoreChart', stats.chartData);

}

}
